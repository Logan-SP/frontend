<script>
  const API_URL = "https://mercadinho-api-k47j.onrender.com";

  function salvarProduto() {
    const id = document.getElementById("produto-id").value;
    const nome = document.getElementById("nome").value;
    const preco = parseFloat(document.getElementById("preco").value);
    const quantidade = parseInt(document.getElementById("quantidade").value);

    const produto = { nome, preco, quantidade };

    if (id) {
      fetch(`${API_URL}/produtos/${id}`, {
        method: "PUT",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(produto)
      }).then(() => {
        alert("Produto atualizado!");
        limparFormulario();
        listarProdutos();
      });
    } else {
      fetch(`${API_URL}/produtos`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(produto)
      }).then(() => {
        alert("Produto criado!");
        limparFormulario();
        listarProdutos();
      });
    }
  }

  function listarProdutos() {
    fetch(`${API_URL}/produtos`)
      .then(res => res.json())
      .then(produtos => {
        const lista = document.getElementById("lista-produtos");
        lista.innerHTML = "";

        produtos.forEach(produto => {
          const li = document.createElement("li");

          li.innerHTML = `
            <div class="produto-info">
              <strong>${produto.nome}</strong><br>
              Preço: R$ ${produto.preco} |
              Quantidade: ${produto.quantidade}
            </div>
            <div class="acoes">
              <button class="btn-warning" onclick="editarProduto(${produto.id}, '${produto.nome}', ${produto.preco}, ${produto.quantidade})">Editar</button>
              <button class="btn-danger" onclick="deletarProduto(${produto.id})">Excluir</button>
            </div>
          `;

          lista.appendChild(li);
        });
      });
  }

  function editarProduto(id, nome, preco, quantidade) {
    document.getElementById("produto-id").value = id;
    document.getElementById("nome").value = nome;
    document.getElementById("preco").value = preco;
    document.getElementById("quantidade").value = quantidade;
    document.getElementById("titulo-form").innerText = "Editar Produto";
  }

  function cancelarEdicao() {
    limparFormulario();
  }

  function limparFormulario() {
    document.getElementById("produto-id").value = "";
    document.getElementById("nome").value = "";
    document.getElementById("preco").value = "";
    document.getElementById("quantidade").value = "";
    document.getElementById("titulo-form").innerText = "Cadastrar Produto";
  }

  function deletarProduto(id) {
    if (!confirm("Deseja realmente excluir o produto?")) return;

    fetch(`${API_URL}/produtos/${id}`, {
      method: "DELETE"
    }).then(() => {
      alert("Produto excluído!");
      listarProdutos();
    });
  }

  listarProdutos();
</script>
